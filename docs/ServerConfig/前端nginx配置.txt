server
{
    listen 80;
    listen 443 ssl http2 ;
    server_name gallery.pinkcandy.top;
    index index.php index.html index.htm default.php default.htm default.html;
    root /www/wwwroot/pinkcandy_gallery;
    include /www/server/panel/vhost/nginx/extension/gallery.pinkcandy.top/*.conf;

    # 需要支持HTTPS SSL......    
    
    # 前端路由管理页面
    location / {
        try_files $uri $uri/ @router;
        index index.html index.htm;
    }
    location @router {
        rewrite ^.*$ /index.html last;
    }
    # 对于后端请求的转发
    location ~ /api/* {
      proxy_pass http://gallery-system.pinkcandy.top;
      rewrite ^/api/(.*)$ /$1 break;
    }
}
